Option Explicit On
Option Strict On

Imports System
Imports System.Data.SqlClient
Imports System.Collections.Generic
Imports Microsoft.SqlServer.Dts.Runtime

' =====================================================================
' PAKET  : Fakten Laden
' SKRIPT : SCR16_Stagingtabelle_Index
' ZWECK  : CI/CCI auf _in_{wert} und _out_{wert} Tabellen erstellen
'          NUR fuer Partitionswerte des aktuellen Laufs (BA::objPartitionValues)
'          Indextyp (CCI/CI) aus Parametertabelle FaktenClusteredIndex
'          Beide Tabellen (_in_ + _out_) erhalten denselben Indextyp
'          da beide fuer den Partitionstausch (SCR19) benoetigt werden
'          Status: PARTITIONLADEN_NACHLAUF_AUSGEFUEHRT -> INDEX_ERSTELLT
' =====================================================================
<Microsoft.SqlServer.Dts.Tasks.ScriptTask.SSISScriptTaskEntryPointAttribute()>
<CLSCompliant(False)>
Partial Public Class ScriptMain
    Inherits Microsoft.SqlServer.Dts.Tasks.ScriptTask.VSTARTScriptObjectModelBase

    Private Const SKRIPT_NAME As String = "SCR16_Stagingtabelle_Index"
    Private Const CONN_NAME As String = "Verbindung"
    Private Const MAX_VERSUCHE As Integer = 10
    Private Const WARTE_SEK As Integer = 30

    Private _runID As Integer = 0
    Private _parameterDB As String = String.Empty
    Private _parametertab As String = String.Empty

    Public Sub Main()
        Log("════════════════════════════════════════════════════════════")
        Log("SCR16_Stagingtabelle_Index - Start")
        Log("Zeitpunkt: " & DateTime.Now.ToString("dd.MM.yyyy HH:mm:ss"))
        Log("════════════════════════════════════════════════════════════")

        Try
            _runID        = Convert.ToInt32(Dts.Variables("BA::RunID").Value)
            _parameterDB  = Dts.Variables("BA::ParameterDB").Value.ToString().Trim()
            _parametertab = Dts.Variables("BA::Parametertabelle").Value.ToString().Trim()
            Dim connStr As String = HoleVerbindungszeichenfolge()

            Log("RunID        : " & _runID.ToString())
            Log("ParameterDB  : " & _parameterDB)

            ' 1. Read partition values from BA::objPartitionValues (set by SCR09)
            Dim partObjekt As Object = Dts.Variables("BA::objPartitionValues").Value
            If partObjekt Is Nothing Then
                Log("BA::objPartitionValues ist leer -> keine Indizes zu erstellen")
                Dts.TaskResult = ScriptResults.Success
                Return
            End If

            Dim partArray(,) As String = CType(partObjekt, String(,))
            Dim anzahlEintraege As Integer = partArray.GetLength(0)
            Log("Partitionswerte aus SCR09: " & anzahlEintraege.ToString() & " Eintraege")

            ' Group partition values by procedure (key = Verfahren.ToLower())
            Dim verfahrenWerte As New Dictionary(Of String, List(Of String))()
            For i As Integer = 0 To anzahlEintraege - 1
                Dim verf As String = partArray(i, 0).Trim().ToLower()
                Dim wert As String = partArray(i, 1).Trim()
                If Not verfahrenWerte.ContainsKey(verf) Then
                    verfahrenWerte(verf) = New List(Of String)()
                End If
                verfahrenWerte(verf).Add(wert)
                Log("  Gelesen: " & verf & " | " & wert)
            Next

            ' 2. Load procedures from work list filtered by RunID
            '    Faktentabelle + FaktenClusteredIndex + Faktenpartitionsspalte
            '    come from the parameter table (NOT from ETL_Fkt_Arbeitsliste)
            Dim verfahren As List(Of VerfahrenInfo) = VerfahrenLaden(connStr)
            Log("Verfahren zur Verarbeitung: " & verfahren.Count.ToString())

            If verfahren.Count = 0 Then
                Log("Keine Verfahren zur Verarbeitung vorhanden.")
                Dts.TaskResult = ScriptResults.Success
                Return
            End If

            Dim cntOK As Integer = 0
            Dim cntFehler As Integer = 0

            For Each v As VerfahrenInfo In verfahren
                Log("────────────────────────────────────────────────────────")
                Log("Verfahren    : " & v.Verfahren)
                Log("Faktentabelle: " & v.Faktentabelle)
                Log("IndexTyp     : " & v.IndexType)
                Log("PartitionSp. : " & v.PartitionSpalte)

                If v.LetzterSchritt = "INDEX_ERSTELLT" Then
                    Log("  -> bereits abgeschlossen -> uebersprungen OK")
                    Continue For
                End If

                Try
                    StatusSetzen(connStr, v.ID, "INDEX_ERSTELLEN")

                    Dim verfKey As String = v.Verfahren.Trim().ToLower()
                    If Not verfahrenWerte.ContainsKey(verfKey) Then
                        Log("  WARNUNG: Keine Partitionswerte fuer '" & v.Verfahren & "' in BA::objPartitionValues -> uebersprungen")
                        Continue For
                    End If

                    Dim werteListe As List(Of String) = verfahrenWerte(verfKey)
                    Log("  Partitionswerte fuer dieses Verfahren: " & werteListe.Count.ToString())

                    ' CCI flag determined from parameter table — same index on _in_ AND _out_
                    ' Both must match the fact table index type for SCR19 partition switch to succeed
                    Dim useCCI As Boolean = (v.IndexType.ToUpper() = "CCI")

                    For Each wert As String In werteListe
                        ' Table names built from Faktentabelle (not Verfahren)
                        ' matching SCR12 naming: [Faktentabelle]_in_[wert] / [Faktentabelle]_out_[wert]
                        Dim inTable  As String = v.Faktentabelle.Trim().ToLower() & "_in_"  & wert
                        Dim outTable As String = v.Faktentabelle.Trim().ToLower() & "_out_" & wert
                        Log("  Partition: " & wert)
                        IndexErstellen(connStr, inTable,  v.PartitionSpalte, useCCI)
                        IndexErstellen(connStr, outTable, v.PartitionSpalte, useCCI)
                    Next

                    StatusSetzen(connStr, v.ID, "INDEX_ERSTELLT")
                    LogSchreiben(connStr, v.Verfahren, "SCHRITT_16",
                        "Indizes (" & v.IndexType & ") auf _in_/_out_ erstellt fuer " &
                        werteListe.Count.ToString() & " Partition(en)")
                    cntOK += 1
                    Log("  -> Indizes erstellt OK")

                Catch ex As Exception
                    cntFehler += 1
                    FehlerSetzen(connStr, v.ID, ex.Message)
                    LogSchreiben(connStr, v.Verfahren, "FEHLER_SCR16", ex.Message)
                    LogFehler("FEHLER bei '" & v.Verfahren & "': " & ex.Message)
                End Try
            Next

            Log("════════════════════════════════════════════════════════════")
            Log("Erfolgreich: " & cntOK.ToString() & " | Fehler: " & cntFehler.ToString())
            Log("════════════════════════════════════════════════════════════")
            Dts.TaskResult = If(cntFehler > 0, ScriptResults.Failure, ScriptResults.Success)

        Catch ex As Exception
            LogFehler("Kritischer Fehler: " & ex.Message)
            Dts.TaskResult = ScriptResults.Failure
        End Try
    End Sub

    ' =========================================================================
    ' Index erstellen — useCCI=True -> CCI, useCCI=False -> CI auf partitionSpalte
    ' Beide _in_ und _out_ erhalten denselben Typ damit Partitionstausch (SCR19)
    ' funktioniert (SQL Server verlangt identischen Clustered Index)
    ' =========================================================================
    Private Sub IndexErstellen(connStr As String, tabellenName As String,
                                partitionSpalte As String, useCCI As Boolean)
        Dim indexName As String = "ix_" & tabellenName

        Dim existsResult As Object = SqlSkalar(connStr,
            "SELECT OBJECT_ID('dbo.[" & tabellenName & "]','U')",
            "Existenzpruefung " & tabellenName)
        If existsResult Is Nothing OrElse existsResult Is DBNull.Value Then
            Log("    WARNUNG: Tabelle dbo.[" & tabellenName & "] nicht gefunden -> uebersprungen")
            Return
        End If

        SqlAusfuehren(connStr,
            "IF EXISTS (SELECT 1 FROM sys.indexes" &
            " WHERE name='" & indexName & "'" &
            " AND object_id=OBJECT_ID('dbo.[" & tabellenName & "]'))" &
            " DROP INDEX [" & indexName & "] ON dbo.[" & tabellenName & "];",
            "Index DROP " & tabellenName)

        Dim sqlCreate As String
        If useCCI Then
            sqlCreate = "CREATE CLUSTERED COLUMNSTORE INDEX [" & indexName & "]" &
                        " ON dbo.[" & tabellenName & "];"
            Log("    CCI angelegt: dbo.[" & tabellenName & "]")
        ElseIf Not String.IsNullOrEmpty(partitionSpalte) Then
            sqlCreate = "CREATE CLUSTERED INDEX [" & indexName & "]" &
                        " ON dbo.[" & tabellenName & "] ([" & partitionSpalte & "]);"
            Log("    CI angelegt : dbo.[" & tabellenName & "] auf [" & partitionSpalte & "]")
        Else
            Log("    WARNUNG: Kein Index fuer dbo.[" & tabellenName &
                "] — kein IndexTyp und keine Partitionsspalte konfiguriert")
            Return
        End If

        SqlAusfuehren(connStr, sqlCreate, "Index CREATE " & tabellenName)
    End Sub

    ' =========================================================================
    ' Verfahren aus ETL_Fkt_Arbeitsliste laden
    ' Faktentabelle, FaktenClusteredIndex, Faktenpartitionsspalte kommen
    ' aus der Parametertabelle (nicht aus ETL_Fkt_Arbeitsliste)
    ' =========================================================================
    Private Function VerfahrenLaden(connStr As String) As List(Of VerfahrenInfo)
        Dim liste As New List(Of VerfahrenInfo)()
        Dim sql As String =
            "SELECT a.ID, a.Verfahren, a.Themengebiet, a.LetzterSchritt," &
            "       pf.Wert  AS Faktentabelle," &
            "       UPPER(pi.Wert) AS IndexType," &
            "       pp.Wert  AS PartitionSpalte" &
            " FROM  dbo.ETL_Fkt_Arbeitsliste a" &
            " JOIN  " & _parameterDB & ".dbo." & _parametertab &
            "       pf ON pf.Verfahren = a.Verfahren AND pf.Parameter = 'Faktentabelle'" &
            " JOIN  " & _parameterDB & ".dbo." & _parametertab &
            "       pi ON pi.Verfahren = a.Verfahren AND pi.Parameter = 'FaktenClusteredIndex'" &
            " JOIN  " & _parameterDB & ".dbo." & _parametertab &
            "       pp ON pp.Verfahren = a.Verfahren AND pp.Parameter = 'Faktenpartitionsspalte'" &
            " WHERE a.Status IN ('PARTITIONLADEN_NACHLAUF_AUSGEFUEHRT','INDEX_ERSTELLEN')" &
            " AND   a.RunID = " & _runID.ToString() &
            " ORDER BY a.Verfahren"

        Dim versuch As Integer = 0
        While versuch < MAX_VERSUCHE
            versuch += 1
            Try
                Using conn As New SqlConnection(connStr)
                    conn.Open()
                    Using cmd As New SqlCommand(sql, conn)
                        cmd.CommandTimeout = 0
                        Log("SQL: " & sql.Substring(0, Math.Min(200, sql.Length)).Replace(vbCrLf, " "))
                        Using rdr As SqlDataReader = cmd.ExecuteReader()
                            While rdr.Read()
                                Dim rawPart As String = rdr(6).ToString().Trim()
                                liste.Add(New VerfahrenInfo With {
                                    .ID             = rdr.GetInt32(0),
                                    .Verfahren      = rdr(1).ToString().Trim(),
                                    .Themengebiet   = rdr(2).ToString().Trim(),
                                    .LetzterSchritt = If(rdr.IsDBNull(3), "", rdr(3).ToString().Trim()),
                                    .Faktentabelle  = rdr(4).ToString().Trim(),
                                    .IndexType      = rdr(5).ToString().Trim(),
                                    .PartitionSpalte = If(rawPart.Contains("|"),
                                                          rawPart.Substring(0, rawPart.IndexOf("|")),
                                                          rawPart)})
                            End While
                        End Using
                    End Using
                End Using
                Return liste
            Catch ex As Exception
                Log(String.Format("WARNUNG [VerfahrenLaden] Versuch {0}/{1}: {2}",
                    versuch, MAX_VERSUCHE, ex.Message))
                If versuch < MAX_VERSUCHE Then System.Threading.Thread.Sleep(WARTE_SEK * 1000) Else Throw
            End Try
        End While
        Return liste
    End Function

    ' =========================================================================
    ' Hilfsmethoden
    ' =========================================================================
    Private Sub StatusSetzen(connStr As String, id As Integer, status As String)
        SqlAusfuehren(connStr,
            "UPDATE dbo.ETL_Fkt_Arbeitsliste" &
            " SET Status='" & status & "', LetzterSchritt='" & status & "', AktualisiertAm=GETDATE()" &
            " WHERE ID=" & id.ToString(), "Status setzen")
    End Sub

    Private Sub FehlerSetzen(connStr As String, id As Integer, msg As String)
        Try
            Using conn As New SqlConnection(connStr)
                conn.Open()
                Using cmd As New SqlCommand(
                    "UPDATE dbo.ETL_Fkt_Arbeitsliste" &
                    " SET Status='FEHLER', Fehlermeldung=@m, AktualisiertAm=GETDATE()" &
                    " WHERE ID=@id", conn)
                    cmd.Parameters.AddWithValue("@m", If(msg.Length > 3900, msg.Substring(0, 3900), msg))
                    cmd.Parameters.AddWithValue("@id", id)
                    cmd.ExecuteNonQuery()
                End Using
            End Using
        Catch
        End Try
    End Sub

    Private Sub LogSchreiben(connStr As String, verfahren As String,
                              schritt As String, meldung As String)
        Try
            Using conn As New SqlConnection(connStr)
                conn.Open()
                Using cmd As New SqlCommand(
                    "INSERT INTO dbo.tm_fakten_load_log(verfahren,schritt,meldung)" &
                    " VALUES(@v,@s,@m)", conn)
                    cmd.Parameters.AddWithValue("@v", verfahren)
                    cmd.Parameters.AddWithValue("@s", schritt)
                    cmd.Parameters.AddWithValue("@m", meldung)
                    cmd.ExecuteNonQuery()
                End Using
            End Using
        Catch
        End Try
    End Sub

    Private Function SqlAusfuehren(connStr As String, sql As String,
                                    beschreibung As String) As Integer
        Dim versuch As Integer = 0
        Dim letzterFehler As Exception = Nothing
        While versuch < MAX_VERSUCHE
            versuch += 1
            Try
                Using conn As New SqlConnection(connStr)
                    conn.Open()
                    Using cmd As New SqlCommand(sql, conn)
                        cmd.CommandTimeout = 0
                        Return cmd.ExecuteNonQuery()
                    End Using
                End Using
            Catch ex As Exception
                letzterFehler = ex
                Log(String.Format("WARNUNG [{0}] Versuch {1}/{2}: {3}",
                    beschreibung, versuch, MAX_VERSUCHE, ex.Message))
                If versuch < MAX_VERSUCHE Then System.Threading.Thread.Sleep(WARTE_SEK * 1000)
            End Try
        End While
        Throw New Exception(String.Format("[{0}] fehlgeschlagen: {1}",
            beschreibung,
            If(letzterFehler IsNot Nothing, letzterFehler.Message, "Unbekannt")))
    End Function

    Private Function SqlSkalar(connStr As String, sql As String,
                                beschreibung As String) As Object
        Dim versuch As Integer = 0
        While versuch < MAX_VERSUCHE
            versuch += 1
            Try
                Using conn As New SqlConnection(connStr)
                    conn.Open()
                    Using cmd As New SqlCommand(sql, conn)
                        cmd.CommandTimeout = 0
                        Return cmd.ExecuteScalar()
                    End Using
                End Using
            Catch ex As Exception
                If versuch < MAX_VERSUCHE Then System.Threading.Thread.Sleep(WARTE_SEK * 1000) Else Throw
            End Try
        End While
        Return Nothing
    End Function

    Private Function HoleVerbindungszeichenfolge() As String
        Return Dts.Connections(CONN_NAME).ConnectionString
    End Function

    Private Sub Log(n As String)
        Dim f As Boolean = False
        Dts.Events.FireInformation(0, SKRIPT_NAME, n, "", 0, f)
    End Sub

    Private Sub LogFehler(n As String)
        Dts.Events.FireError(0, SKRIPT_NAME, n, "", 0)
    End Sub

    Private Class VerfahrenInfo
        Public Property ID As Integer
        Public Property Verfahren As String
        Public Property Themengebiet As String
        Public Property LetzterSchritt As String
        Public Property Faktentabelle As String
        Public Property IndexType As String
        Public Property PartitionSpalte As String
    End Class

    Public Enum ScriptResults
        Success = DTSExecResult.Success
        Failure = DTSExecResult.Failure
    End Enum

End Class
