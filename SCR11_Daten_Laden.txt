Option Explicit On
Option Strict On

Imports System
Imports System.Data.SqlClient
Imports System.Collections.Generic
Imports System.Collections.Concurrent
Imports System.Text
Imports System.Threading
Imports System.Threading.Tasks
Imports System.Linq
Imports Microsoft.SqlServer.Dts.Runtime

' =============================================================================
' PAKET  : Fakten Laden
' SKRIPT : SCR11_Daten_Laden
' ZWECK  : PARALLELES Datenladen aller _in Partitionen mit SELECT INTO
'          Status: STAGING_ERSTELLT -> DATEN_GELADEN
' =============================================================================
<Microsoft.SqlServer.Dts.Tasks.ScriptTask.SSISScriptTaskEntryPointAttribute()>
<CLSCompliant(False)>
Partial Public Class ScriptMain
    Inherits Microsoft.SqlServer.Dts.Tasks.ScriptTask.VSTARTScriptObjectModelBase

    Private Const SKRIPT_NAME As String = "SCR11_Daten_Laden"
    Private Const CONN_NAME As String = "Verbindung"
    Private Const MAX_VERSUCHE As Integer = 10
    Private Const WARTE_SEK As Integer = 30

    Private _runID As Integer = 0
    Private _extTableSchema As String = String.Empty
    Private _maxParallel As Integer = 4

    Private ReadOnly _logSperre As New Object()
    Private ReadOnly _fehlerListe As New ConcurrentBag(Of String)()
    Private _cntOK As Integer = 0
    Private _cntFehler As Integer = 0

    Public Sub Main()
        Log("=============================================================================")
        Log("SCR11_Daten_Laden (Parallel Partition Load) - Start")
        Log("Zeitpunkt: " & DateTime.Now.ToString("dd.MM.yyyy HH:mm:ss"))
        Log("=============================================================================")

        Try
            _runID = Convert.ToInt32(Dts.Variables("BA::RunID").Value)
            _extTableSchema = Dts.Variables("BA::ExtTableSchema").Value.ToString().Trim()
            
            ' Try to get MaxParallel from variable, fallback to 4
            Try
                _maxParallel = Convert.ToInt32(Dts.Variables("BA::MaxParallel").Value)
            Catch
                _maxParallel = 4
            End Try

            Log("Ext. Schema: " & _extTableSchema)
            Log("Parallelitaet: " & _maxParallel.ToString())

            Dim connStr As String = HoleVerbindungszeichenfolge()
            Dim verfahren As List(Of VerfahrenInfo) = VerfahrenLaden(connStr)
            Log("Verfahren zur Verarbeitung: " & verfahren.Count.ToString())

            If verfahren.Count = 0 Then
                Log("Keine Verfahren zur Verarbeitung vorhanden.")
                Dts.TaskResult = ScriptResults.Success
                Return
            End If

            ' Parallel loop over Verfahren
            Dim opts As New ParallelOptions With {.MaxDegreeOfParallelism = _maxParallel}
            Parallel.ForEach(verfahren, opts, Sub(v)
                Try
                    VerarbeiteVerfahren(v, connStr)
                Catch ex As Exception
                    _fehlerListe.Add("KRITISCHER FEHLER [" & v.Verfahren & "]: " & ex.Message)
                    Interlocked.Increment(_cntFehler)
                End Try
            End Sub)

            Log("=============================================================================")
            Log("Erfolgreich: " & _cntOK.ToString() & " | Fehler: " & _cntFehler.ToString())
            
            If _fehlerListe.Count > 0 Then
                Log("FEHLERLISTE:")
                For Each msg As String In _fehlerListe.OrderBy(Function(x) x)
                    Log("  - " & msg)
                Next
            End If
            Log("=============================================================================")

            Dts.TaskResult = If(_cntFehler > 0, ScriptResults.Failure, ScriptResults.Success)
        Catch ex As Exception
            LogFehler("Kritischer Fehler in Main: " & ex.Message)
            Dts.TaskResult = ScriptResults.Failure
        End Try
    End Sub

    Private Sub VerarbeiteVerfahren(v As VerfahrenInfo, connStr As String)
        SyncLock _logSperre
            Log("------------------------------------------------------------")
            Log("Verfahren: " & v.Verfahren & " | Themengebiet: " & v.Themengebiet)
        End SyncLock

        If v.LetzterSchritt = "DATEN_GELADEN" Then
            Log("  -> bereits abgeschlossen -> uebersprungen ok")
            Interlocked.Increment(_cntOK)
            Return
        End If

        Try
            StatusSetzen(connStr, v.ID, "DATEN_LADEN")

            ' Find all _in tables for this fact table
            Dim inTables As List(Of String) = InTabellenSuchen(connStr, v.Verfahren)
            Log("  Gefundene Partitionen (_in): " & inTables.Count.ToString())

            Dim totalRows As Long = 0
            
            ' Process each partition
            For Each inTable As String In inTables
                Dim rows As Integer = PartitionLaden(connStr, v, inTable)
                totalRows += rows
            Next

            StatusSetzen(connStr, v.ID, "DATEN_GELADEN")
            LogSchreiben(connStr, v.Verfahren, "SCHRITT_11", "Daten geladen: " & totalRows.ToString() & " Zeilen in " & inTables.Count.ToString() & " Partitionen")
            
            Interlocked.Increment(_cntOK)
            Log("  -> " & totalRows.ToString() & " Zeilen geladen ok")

        Catch ex As Exception
            Interlocked.Increment(_cntFehler)
            _fehlerListe.Add("FEHLER [" & v.Verfahren & "]: " & ex.Message)
            FehlerSetzen(connStr, v.ID, ex.Message)
            LogSchreiben(connStr, v.Verfahren, "FEHLER_SCR11", ex.Message)
            LogFehler("FEHLER bei '" & v.Verfahren & "': " & ex.Message)
        End Try
    End Sub

    Private Function PartitionLaden(connStr As String, v As VerfahrenInfo, inTable As String) As Integer
        Dim extTabName As String = v.Verfahren.Trim().ToUpper()
        Dim partitionValue As String = inTable.Split("_"c).Last()
        
        ' 1. Check if already loaded
        Dim abgeschlossenZeilen As Integer = -1
        Try
            Dim result As Object = SqlSkalar(connStr,
                "SELECT TOP 1 Zeilen FROM dbo.ETL_Fakt_Laden_Abschluss WHERE InTabelle = '" & inTable & "' ORDER BY AbgeschlossenAm DESC",
                "Abschluss Check")
            If result IsNot Nothing AndAlso result IsNot DBNull.Value Then
                abgeschlossenZeilen = Convert.ToInt32(result)
            End If
        Catch
        End Try

        If abgeschlossenZeilen >= 0 Then
            Log("    " & inTable & ": bereits geladen (" & abgeschlossenZeilen.ToString() & ") ok")
            Return abgeschlossenZeilen
        End If

        ' 2. Build Mapping
        Dim sbCols As New StringBuilder()
        Using conn As New SqlConnection(connStr)
            conn.Open()
            Using cmd As New SqlCommand("SELECT columns_dbo FROM dbo.tm_polybase_struktur WHERE themengebiet=@thema AND tabname=@tab ORDER BY colno;", conn)
                cmd.Parameters.AddWithValue("@thema", v.Themengebiet.Trim().ToLower())
                cmd.Parameters.AddWithValue("@tab", v.Verfahren.Trim().ToLower())
                Using rdr As SqlDataReader = cmd.ExecuteReader()
                    While rdr.Read()
                        If sbCols.Length > 0 Then sbCols.Append("," & Environment.NewLine)
                        sbCols.Append(rdr(0).ToString())
                    End While
                End Using
            End Using
        End Using

        If sbCols.Length = 0 Then Throw New Exception("Keine Spalten-Truth in tm_polybase_struktur fuer " & v.Verfahren)

        ' 3. Drop and SELECT INTO
        SqlAusfuehren(connStr, "IF OBJECT_ID('dbo.[" & inTable & "]','U') IS NOT NULL DROP TABLE dbo.[" & inTable & "];", "DROP " & inTable)

        Dim sqlSelectInto As String =
"SELECT" & Environment.NewLine &
sbCols.ToString() & Environment.NewLine &
"INTO dbo.[" & inTable & "]" & Environment.NewLine &
"FROM [" & _extTableSchema & "].[" & extTabName & "]" & Environment.NewLine &
"WHERE [" & v.PartitionColumn & "] = " & partitionValue & ";"

        Dim affectedRows As Integer = SqlAusfuehren(connStr, sqlSelectInto, "SELECT INTO " & inTable)
        
        ' If SELECT INTO returns -1 (common for DDL), count manually
        If affectedRows < 0 Then
            affectedRows = Convert.ToInt32(SqlSkalar(connStr, "SELECT COUNT(*) FROM dbo.[" & inTable & "];", "Count " & inTable))
        End If

        ' 4. Log completion
        Try
            SqlAusfuehren(connStr, 
                "INSERT INTO dbo.ETL_Fakt_Laden_Abschluss (RunID, Verfahren, InTabelle, Zeilen) VALUES (" & _runID & ", '" & v.Verfahren & "', '" & inTable & "', " & affectedRows & ")",
                "Abschluss schreiben")
        Catch
        End Try

        Log("    " & inTable & ": " & affectedRows.ToString() & " Zeilen geladen ok")
        Return affectedRows
    End Function

    Private Function InTabellenSuchen(connStr As String, verfahren As String) As List(Of String)
        Dim liste As New List(Of String)()
        Try
            Using conn As New SqlConnection(connStr)
                conn.Open()
                Using cmd As New SqlCommand("SELECT name FROM sys.tables WHERE schema_id=SCHEMA_ID('dbo') AND name LIKE '" & verfahren.ToLower() & "_in_%' ORDER BY name", conn)
                    Using rdr As SqlDataReader = cmd.ExecuteReader()
                        While rdr.Read()
                            liste.Add(rdr(0).ToString())
                        End While
                    End Using
                End Using
            End Using
        Catch
        End Try
        Return liste
    End Function

    Private Function VerfahrenLaden(connStr As String) As List(Of VerfahrenInfo)
        Dim liste As New List(Of VerfahrenInfo)()
        ' Note: Using ETL_Fkt_Arbeitsliste as driver, but need partition col from parameters
        ' If ETL_Fkt_Parameter table doesn't exist, we fallback or join appropriately
        Dim sql As String =
"SELECT a.ID, a.Verfahren, a.Themengebiet, a.LetzterSchritt, 
        (SELECT TOP 1 Wert FROM dbo.ETL_Fkt_Parameter WHERE Verfahren = a.Verfahren AND Parameter = 'Faktenpartitionsspalte') as PartitionColumn
 FROM dbo.ETL_Fkt_Arbeitsliste a
 WHERE a.Status IN ('STAGING_ERSTELLT','DATEN_LADEN') AND a.RunID=" & _runID.ToString() & " ORDER BY a.Verfahren"
        
        Using conn As New SqlConnection(connStr)
            conn.Open()
            Using cmd As New SqlCommand(sql, conn)
                Using rdr As SqlDataReader = cmd.ExecuteReader()
                    While rdr.Read()
                        liste.Add(New VerfahrenInfo With {
                            .ID = rdr.GetInt32(0), .Verfahren = rdr(1).ToString().Trim(),
                            .Themengebiet = rdr(2).ToString().Trim(),
                            .LetzterSchritt = If(rdr.IsDBNull(3), "", rdr(3).ToString().Trim()),
                            .PartitionColumn = If(rdr.IsDBNull(4), "partition_id", rdr(4).ToString().Trim())})
                    End While
                End Using
            End Using
        End Using
        Return liste
    End Function

    Private Sub StatusSetzen(connStr As String, id As Integer, status As String)
        SqlAusfuehren(connStr, "UPDATE dbo.ETL_Fkt_Arbeitsliste SET Status='" & status & "',LetzterSchritt='" & status & "',AktualisiertAm=GETDATE() WHERE ID=" & id.ToString(), "Status setzen")
    End Sub

    Private Sub FehlerSetzen(connStr As String, id As Integer, msg As String)
        Try
            Using conn As New SqlConnection(connStr)
                conn.Open()
                Using cmd As New SqlCommand("UPDATE dbo.ETL_Fkt_Arbeitsliste SET Status='FEHLER',Fehlermeldung=@m,AktualisiertAm=GETDATE() WHERE ID=@id", conn)
                    cmd.Parameters.AddWithValue("@m", If(msg.Length > 3900, msg.Substring(0, 3900), msg))
                    cmd.Parameters.AddWithValue("@id", id)
                    cmd.ExecuteNonQuery()
                End Using
            End Using
        Catch
        End Try
    End Sub

    Private Sub LogSchreiben(connStr As String, verfahren As String, schritt As String, meldung As String)
        Try
            Using conn As New SqlConnection(connStr)
                conn.Open()
                Using cmd As New SqlCommand("INSERT INTO dbo.tm_fakten_load_log(verfahren,schritt,meldung) VALUES(@v,@s,@m)", conn)
                    cmd.Parameters.AddWithValue("@v", verfahren)
                    cmd.Parameters.AddWithValue("@s", schritt)
                    cmd.Parameters.AddWithValue("@m", meldung)
                    cmd.ExecuteNonQuery()
                End Using
            End Using
        Catch
        End Try
    End Sub

    Private Function SqlAusfuehren(connStr As String, sql As String, beschreibung As String) As Integer
        Dim versuch As Integer = 0
        Dim letzterFehler As Exception = Nothing
        While versuch < MAX_VERSUCHE
            versuch += 1
            Try
                Using conn As New SqlConnection(connStr)
                    conn.Open()
                    Using cmd As New SqlCommand(sql, conn)
                        cmd.CommandTimeout = 0
                        Return cmd.ExecuteNonQuery()
                    End Using
                End Using
            Catch ex As Exception
                letzterFehler = ex
                If versuch < MAX_VERSUCHE Then System.Threading.Thread.Sleep(WARTE_SEK * 1000)
            End Try
        End While
        Throw New Exception(String.Format("[{0}] fehlgeschlagen: {1}", beschreibung, If(letzterFehler IsNot Nothing, letzterFehler.Message, "Unbekannt")))
    End Function

    Private Function SqlSkalar(connStr As String, sql As String, beschreibung As String) As Object
        Dim versuch As Integer = 0
        While versuch < MAX_VERSUCHE
            versuch += 1
            Try
                Using conn As New SqlConnection(connStr)
                    conn.Open()
                    Using cmd As New SqlCommand(sql, conn)
                        cmd.CommandTimeout = 0
                        Return cmd.ExecuteScalar()
                    End Using
                End Using
            Catch ex As Exception
                If versuch < MAX_VERSUCHE Then System.Threading.Thread.Sleep(WARTE_SEK * 1000) Else Throw
            End Try
        End While
        Return Nothing
    End Function

    Private Function HoleVerbindungszeichenfolge() As String
        Return Dts.Connections(CONN_NAME).ConnectionString
    End Function

    Private Sub Log(n As String)
        SyncLock _logSperre
            Dim f As Boolean = False
            Dts.Events.FireInformation(0, SKRIPT_NAME, n, "", 0, f)
        End SyncLock
    End Sub

    Private Sub LogFehler(n As String)
        SyncLock _logSperre
            Dts.Events.FireError(0, SKRIPT_NAME, n, "", 0)
        End SyncLock
    End Sub

    Private Class VerfahrenInfo
        Public Property ID As Integer
        Public Property Verfahren As String
        Public Property Themengebiet As String
        Public Property LetzterSchritt As String
        Public Property PartitionColumn As String
    End Class

    Public Enum ScriptResults
        Success = DTSExecResult.Success
        Failure = DTSExecResult.Failure
    End Enum

End Class
