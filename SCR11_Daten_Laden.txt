    Private Function DatenLaden(connStr As String, thema As String, tab As String) As Integer
        Dim stgInName As String = tab.Trim().ToLower() & "_in"
        Dim extTabName As String = tab.Trim().ToUpper()

        ' â”€â”€ COMPLETION TRACKING CHECK â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
        ' Check ETL_Fakt_Laden_Abschluss â€” this is written ONLY when loading
        ' fully completes for this _in table. COUNT(*) alone is not reliable
        ' because a crashed load can leave _in partially filled.
        '
        ' Entry exists   â†’ fully loaded previously â†’ SKIP
        ' No entry       â†’ never loaded OR partially loaded â†’ TRUNCATE + RELOAD
        ' â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
        Dim abgeschlossenZeilen As Integer = 0
        Try
            Dim sqlAbschluss As String =
                "SELECT TOP 1 Zeilen FROM dbo.ETL_Fakt_Laden_Abschluss " &
                "WHERE InTabelle = @tbl ORDER BY AbgeschlossenAm DESC"
            Dim result As Object = SqlSkalar(connStr,
                sqlAbschluss.Replace("@tbl", "'" & stgInName & "'"),
                "Abschluss Check")
            If result IsNot Nothing AndAlso result IsNot DBNull.Value Then
                abgeschlossenZeilen = Convert.ToInt32(result)
            End If
        Catch
            abgeschlossenZeilen = 0
        End Try

        If abgeschlossenZeilen > 0 Then
            ' Also verify _in table actually still exists (PARTITIONSTAUSCH may have dropped it)
            Dim tabelleExistiert As Integer = 0
            Try
                tabelleExistiert = Convert.ToInt32(SqlSkalar(connStr,
                    "SELECT CASE WHEN OBJECT_ID('dbo." & stgInName & "','U') IS NOT NULL THEN 1 ELSE 0 END",
                    "Tabelle existiert"))
            Catch
            End Try

            If tabelleExistiert = 1 Then
                Log("  CHECKPOINT: dbo.[" & stgInName & "] vollstÃ¤ndig geladen " &
                    "(" & abgeschlossenZeilen.ToString() & " Zeilen) â†’ DATEN_LADEN Ã¼bersprungen âœ“")
                Return abgeschlossenZeilen
            Else
                ' Table was dropped by PARTITIONSTAUSCH â€” clear record and reload
                Log("  HINWEIS: Abschluss-Eintrag vorhanden, aber _in Tabelle gelÃ¶scht " &
                    "(PARTITIONSTAUSCH) â†’ Abschluss-Eintrag lÃ¶schen + neu laden")
                Try
                    SqlAusfuehren(connStr,
                        "DELETE FROM dbo.ETL_Fakt_Laden_Abschluss WHERE InTabelle='" & stgInName & "'",
                        "Abschluss lÃ¶schen")
                Catch
                End Try
                abgeschlossenZeilen = 0  ' Fall through to reload
            End If
        End If

        ' No completion record â†’ table might be partially filled â†’ TRUNCATE first
        Log("  Kein Abschluss-Eintrag â†’ truncate _in (falls partial) + lade neu")
        Try
            If Convert.ToInt32(SqlSkalar(connStr,
                "SELECT CASE WHEN OBJECT_ID('dbo." & stgInName & "','U') IS NOT NULL THEN 1 ELSE 0 END",
                "exist check")) = 1 Then
                SqlAusfuehren(connStr, "TRUNCATE TABLE dbo.[" & stgInName & "];", "TRUNCATE partial _in")
                Log("  _in truncated (war mÃ¶glicherweise teilweise befÃ¼llt)")
            End If
        Catch
        End Try
        ' â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

        ' Load fresh from Oracle
        Log("  Lade Daten aus Oracle...")

        ' Build column mapping from tm_polybase_struktur (ISNULL/CONVERT expressions needed for Oracleâ†’SQL Server casting)
        Dim sbCols As New StringBuilder()
        Dim sqlCols As String =
"SELECT columns_dbo FROM dbo.tm_polybase_struktur
 WHERE themengebiet=@thema AND tabname=@tab ORDER BY colno;"
        Using conn As New SqlConnection(connStr)
            conn.Open()
            Using cmd As New SqlCommand(sqlCols, conn)
                cmd.CommandTimeout = 0
                cmd.Parameters.AddWithValue("@thema", thema.Trim().ToLower())
                cmd.Parameters.AddWithValue("@tab", tab.Trim().ToLower())
                Using rdr As SqlDataReader = cmd.ExecuteReader()
                    While rdr.Read()
                        If sbCols.Length > 0 Then sbCols.Append("," & Environment.NewLine)
                        sbCols.Append(rdr(0).ToString())
                    End While
                End Using
            End Using
        End Using
        If sbCols.Length = 0 Then Throw New Exception("Keine Spalten in tm_polybase_struktur fÃ¼r " & thema & "." & tab)

        ' SELECT INTO staging _in from external table
        ' Drop table if exists because SELECT INTO creates it
        SqlAusfuehren(connStr, "IF OBJECT_ID('dbo.[" & stgInName & "]','U') IS NOT NULL DROP TABLE dbo.[" & stgInName & "];", "DROP " & stgInName)

        Dim sqlSelectInto As String =
"SELECT" & Environment.NewLine &
sbCols.ToString() & Environment.NewLine &
"INTO dbo.[" & stgInName & "]" & Environment.NewLine &
"FROM [" & _extTableSchema & "].[" & extTabName & "];"

        Dim affectedRows As Integer = SqlAusfuehren(connStr, sqlSelectInto, "Daten SELECT INTO")
        If affectedRows < 0 Then
            affectedRows = Convert.ToInt32(SqlSkalar(connStr,
                "SELECT COUNT(*) FROM dbo.[" & stgInName & "];", "Staging _in COUNT"))
        End If

        ' â”€â”€ WRITE COMPLETION RECORD â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
        ' Only written AFTER full successful load.
        ' Crash during load = no record = retry will TRUNCATE + RELOAD this _in.
        Try
            Using conn As New SqlConnection(connStr)
                conn.Open()
                Using cmd As New SqlCommand(
"INSERT INTO dbo.ETL_Fakt_Laden_Abschluss (RunID, Verfahren, InTabelle, Zeilen)
 VALUES (@runID, @verf, @tbl, @zeilen)", conn)
                    cmd.Parameters.AddWithValue("@runID",  _runID)
                    cmd.Parameters.AddWithValue("@verf",   thema.Trim().ToLower() & "." & tab.Trim().ToLower())
                    cmd.Parameters.AddWithValue("@tbl",    stgInName)
                    cmd.Parameters.AddWithValue("@zeilen", affectedRows)
                    cmd.ExecuteNonQuery()
                End Using
            End Using
            Log("  Abschluss-Eintrag geschrieben: " & stgInName & " (" & affectedRows.ToString() & " Zeilen) âœ“")
        Catch ex As Exception
            Log("  WARNUNG: Abschluss-Eintrag konnte nicht geschrieben werden: " & ex.Message)
        End Try
        ' â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

        Log("  " & affectedRows.ToString() & " Zeilen in dbo.[" & stgInName & "] geladen.")
        Return affectedRows
    End Function

    Private Function VerfahrenLaden(connStr As String) As List(Of VerfahrenInfo)
        Dim liste As New List(Of VerfahrenInfo)()
        Dim sql As String =
"SELECT ID, Verfahren, Themengebiet, LetzterSchritt FROM dbo.ETL_Fkt_Arbeitsliste
 WHERE Status IN ('STAGING_ERSTELLT','DATEN_LADEN') AND RunID=" & _runID.ToString() & " ORDER BY Verfahren"
        Dim versuch As Integer = 0
        While versuch < MAX_VERSUCHE
            versuch += 1
            Try
                Using conn As New SqlConnection(connStr)
                    conn.Open()
                    Using cmd As New SqlCommand(sql, conn)
                        cmd.CommandTimeout = 0
                        Using rdr As SqlDataReader = cmd.ExecuteReader()
                            While rdr.Read()
                                liste.Add(New VerfahrenInfo With {
                                    .ID = rdr.GetInt32(0), .Verfahren = rdr(1).ToString().Trim(),
                                    .Themengebiet = rdr(2).ToString().Trim(),
                                    .LetzterSchritt = If(rdr.IsDBNull(3), "", rdr(3).ToString().Trim())})
                            End While
                        End Using
                    End Using
                End Using
                Return liste
            Catch ex As Exception
                Log(String.Format("WARNUNG [Verfahren laden] Versuch {0}/{1}: {2}", versuch, MAX_VERSUCHE, ex.Message))
                If versuch < MAX_VERSUCHE Then System.Threading.Thread.Sleep(WARTE_SEK * 1000) Else Throw
            End Try
        End While
        Return liste
    End Function

    Private Sub StatusSetzen(connStr As String, id As Integer, status As String)
        SqlAusfuehren(connStr, "UPDATE dbo.ETL_Fkt_Arbeitsliste SET Status='" & status & "',LetzterSchritt='" & status & "',AktualisiertAm=GETDATE() WHERE ID=" & id.ToString(), "Status setzen")
    End Sub

    Private Sub FehlerSetzen(connStr As String, id As Integer, msg As String)
        Try
            Using conn As New SqlConnection(connStr)
                conn.Open()
                Using cmd As New SqlCommand("UPDATE dbo.ETL_Fkt_Arbeitsliste SET Status='FEHLER',Fehlermeldung=@m,AktualisiertAm=GETDATE() WHERE ID=@id", conn)
                    cmd.Parameters.AddWithValue("@m", If(msg.Length > 3900, msg.Substring(0, 3900), msg))
                    cmd.Parameters.AddWithValue("@id", id)
                    cmd.ExecuteNonQuery()
                End Using
            End Using
        Catch
        End Try
    End Sub

    Private Sub LogSchreiben(connStr As String, verfahren As String, schritt As String, meldung As String)
        Try
            Using conn As New SqlConnection(connStr)
                conn.Open()
                Using cmd As New SqlCommand("INSERT INTO dbo.tm_fakten_load_log(verfahren,schritt,meldung) VALUES(@v,@s,@m)", conn)
                    cmd.Parameters.AddWithValue("@v", verfahren)
                    cmd.Parameters.AddWithValue("@s", schritt)
                    cmd.Parameters.AddWithValue("@m", meldung)
                    cmd.ExecuteNonQuery()
                End Using
            End Using
        Catch
        End Try
    End Sub

    Private Function SqlAusfuehren(connStr As String, sql As String, beschreibung As String) As Integer
        Dim versuch As Integer = 0
        Dim letzterFehler As Exception = Nothing
        While versuch < MAX_VERSUCHE
            versuch += 1
            Try
                Using conn As New SqlConnection(connStr)
                    conn.Open()
                    Using cmd As New SqlCommand(sql, conn)
                        cmd.CommandTimeout = 0
                        Return cmd.ExecuteNonQuery()
                    End Using
                End Using
            Catch ex As Exception
                letzterFehler = ex
                Log(String.Format("WARNUNG [{0}] Versuch {1}/{2}: {3}", beschreibung, versuch, MAX_VERSUCHE, ex.Message))
                If versuch < MAX_VERSUCHE Then System.Threading.Thread.Sleep(WARTE_SEK * 1000)
            End Try
        End While
        Throw New Exception(String.Format("[{0}] fehlgeschlagen: {1}", beschreibung, If(letzterFehler IsNot Nothing, letzterFehler.Message, "Unbekannt")))
    End Function

    Private Function SqlSkalar(connStr As String, sql As String, beschreibung As String) As Object
        Dim versuch As Integer = 0
        While versuch < MAX_VERSUCHE
            versuch += 1
            Try
                Using conn As New SqlConnection(connStr)
                    conn.Open()
                    Using cmd As New SqlCommand(sql, conn)
                        cmd.CommandTimeout = 0
                        Return cmd.ExecuteScalar()
                    End Using
                End Using
            Catch ex As Exception
                Log(String.Format("WARNUNG [{0}] Versuch {1}/{2}: {3}", beschreibung, versuch, MAX_VERSUCHE, ex.Message))
                If versuch < MAX_VERSUCHE Then System.Threading.Thread.Sleep(WARTE_SEK * 1000) Else Throw
            End Try
        End While
        Return Nothing
    End Function

    Private Function HoleVerbindungszeichenfolge() As String

    End Function

    Private Sub Log(n As String)

    End Sub

    Private Sub LogFehler(n As String)
        Dts.Events.FireError(0, SKRIPT_NAME, n, "", 0)
    End Sub
